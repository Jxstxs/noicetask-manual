101void get_database_results(vector *results, sqlite3 **db, const char *sql_stmt) {
102 int rc;
103 sqlite3_stmt *stmt;
104
105 rc = sqlite3_prepare_v2(*db, sql_stmt, -1, &stmt, 0);
106 if (rc != SQLITE_OK) {
107 printf("!SQL ERROR: %s\n", sqlite3_errmsg(*db));
108 sqlite3_close(*db);
109 exit(EXIT_FAILURE);
110 }
111
112 int columnCount;
113 char *columnName;
114 database_result *row_result;
115 while (sqlite3_step(stmt) == SQLITE_ROW) {
116 // get data from row
117 init_database_result(&row_result);
118
119 columnCount = sqlite3_column_count(stmt);
120 for (int i = 0; i < columnCount; i++) {
121 columnName = (char *)sqlite3_column_name(stmt, i);
122 if (STR_COMPARE(columnName, "tag_id"))
123 row_result->tag_id = sqlite3_column_int(stmt, i);
124 if (STR_COMPARE(columnName, "tag_name"))
125 snprintf(row_result->tag_name, sizeof(row_result->tag_name), "%s", sqlite3_column_text(stmt, i));
126 if (STR_COMPARE(columnName, "tag_desc"))
127 snprintf(row_result->tag_desc, sizeof(row_result->tag_desc), "%s", sqlite3_column_text(stmt, i));
128 }
129 results->pfVectorAdd(results, (void *)row_result);
130 }
131
132 sqlite3_finalize(stmt);
133}
